start_file=start_x.elf
fixup_file=fixup_x.dat

# Kernel location
kernel=zImage

# How much memory in MB to assign to the GPU on Pi models having
gpu_mem=128

# Enable Camera module
start_x=1

# Set the bootloader delay to 0 seconds. The default is 1s if not specified
boot_delay=0
disable_splash=1
force_turbo=1

# DPI Display
ignore_lcd=0
hdmi_ignore_hotplug=1
disable_hdmi=1
gpio=0-9=a2
gpio=12-17=a2
gpio=20-25=a2
enable_dpi_lcd=1
display_default_lcd=1
dpi_group=2
dpi_mode=87
dpi_output_format=0x7F206
hdmi_timings=480 0 26 16 10 640 0 25 10 15 0 0 0 60 0 32000000 1
extra_transpose_buffer=2

# Overlays
dtoverlay=waveshare-28dpi-3b-4b
dtoverlay=waveshare-28dpi-3b
dtoverlay=waveshare-touch-28dpi
dtparam=i2c_arm=on
dtparam=spi=off
dtparam=audio=off

# raspberry firmware config to disable wifi and bluetooth (Needed or Pi0w)
# https://github.com/raspberrypi/firmware/blob/master/boot/overlays/README
dtoverlay=disable-wifi
dtoverlay=disable-bt

# Use bcm2835_mmc rather than sdhost for MicroSD driver (Doesn't abusively poll when MicroSD removed)
dtoverlay=mmc

# ***********************
# Note regarding Wifi
# ***********************
# If you intend to to use wifi, you will need to comment out (or delete) two lines above.
# dtoverlay=disable-wifi
# dtoverlay=mmc
# The first needs to be removed so that wifi isn't disabled.
# The second needs to be removed so that the system defaults to the normal SDIO driver.
# This means that if you are using Wifi and the smartcard hat, the smartcard hat will probably stop working while the MicroSD card is removed.
